# summarize the data with our package
selected_vars <- c("y0age", "y1sex", "y1gen", "y1rac1", "y1rac2", "y1rac3", "y1rac4", "y1rac5", "y1rac999", "y0scr06", "y1eco")
dem_data <- subset(data, select = selected_vars)

# Factor to see labels
dem_data$y0age <- as.numeric(dem_data$y0age)

dem_data <- dem_data %>%
  mutate(across(c(y1sex, y1gen, y1rac1, y1rac2, y1rac3, y1rac4, y1rac5, y1rac999, y0scr06, y1eco), ~haven::as_factor(.)))

table1 <-
